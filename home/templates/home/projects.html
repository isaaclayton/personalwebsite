Abstract¶

Jeopardy is one of my favorite shows and was a daily part of my family's life, so I thought it would be interesting to make discoveries about its questions and contestants. Using Regular Expressions and BeautifulSoup, I went through J-Archive.com, a website that archives every Jeopardy episode, I was able to scrape together a dataset that incorporated about 13 factors regarding each question. This process took incredibly long and while I wasn't able to get to modeling and machine learning, I learned a lot about regular expressions.

Description¶

All together there were 13 factors in the data: Show number, season, the date it aired, comments about the episode, contestant names, occupations, locations, how much they had before Final Jeopardy, how much they had after Final Jeopardy, the round of Jeopardy (Jeopardy, Double Jeopardy, or Final Jeopardy), the question, the answer, and how much the question was worth.

def get_ep(url):
    soup = BeautifulSoup(requests.get(url).text, 'html.parser')
    show_num = re.findall('show (#[0-9]+)\.?',str(soup))
    contestant_info = [list(b) for c in [contestant for a in soup.find_all('p', class_='contestants')
                       for contestant in [re.findall(r'"external">(.+)</a>, (?:a |an )(.+)from([A-Za-z \,]+)(?:.*)?</p>', 
                        str(a))]] for b in c]
    round1 = list(zip(re.findall('class="category_name">(.+)</td>', str(soup.find_all('div', id='jeopardy_round'))), 
            [re.findall(r'onmouseout.*?clue_J_' + str(i) + 
           r"_[0-9]_stuck\', \'(.*?)\'\).*correct_response&quot;&gt;(.*?)&lt;/em&gt;(?:.|\n)*?\"clue_value\">(\$?\d+)" , 
           str(soup.find_all('div', id='jeopardy_round'))) for i in range(1,7)]))
    round1 = [['Jeopardy',cat[0]] + list(q) for cat in [list(a) for a in round1] for q in cat[1]]
    round2 = list(zip(re.findall('class="category_name">(.+)</td>', str(soup.find_all('div', id='double_jeopardy_round'))), 
            [re.findall(r'onmouseout.*?clue_DJ_' + str(i) + 
           r"_[0-9]_stuck\', \'(.*?)\'\).*correct_response&quot;&gt;(.*?)&lt;/em&gt;(?:.|\n)*?\"clue_value\">(\$?\d+)" , 
           str(soup.find_all('div', id='double_jeopardy_round'))) for i in range(1,7)]))
    round2 = [['Double Jeopardy',cat[0]] + list(q) for cat in [list(a) for a in round2] for q in cat[1]]    
    final = [group for innergroup in [[a[-1],list(a[:-1])] for a in re.findall(r'onmouseout.*?clue_FJ_stuck\', \'(.*?)\'\).' + 
            r'*correct_response\\&quot;&gt;(.*?)&lt;/em&gt;(?:.|\n)*category_name">(.*)</td>',
            str(soup.find_all('div', id='final_jeopardy_round')))] for group in innergroup]
    if len(final)>0:
        final = ['Final Jeopardy',final[0]] + final[1] + ['0']
    results = [list(contestant) for group in [list(zip(re.findall(r'nickname">(.+)</td',endscores),
            re.findall(r'(?:positive|negative)">([-$\d,]+)</td>',fjscores),
            re.findall(r'(?:positive|negative)">([-$\d,]+)</td>',endscores))) 
            for endscores in re.findall('Final scores\:((?:.|\n)*?)</table>', 
            str(soup)) for fjscores in 
            re.findall('Scores at the end of the Double Jeopardy\! Round\:((?:.|\n)*?)</table>', 
            str(soup))] for contestant in group]
    contestants = ["%%".join(contestant) for contestant in contestant_info] + ["%%".join(contestant) for contestant in results]
    return [show_num + contestants+question for question in round1+round2+[final]]
